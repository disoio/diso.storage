  • ¶

    ttl : override global ttl

      _expires : (ttl)->
    unless (ttl > 0)
      ttl = @ttl

    if (ttl > 0)
      now() + ttl
    else
      null

      unwrap : (args)->
    {wrapped, key} = args
    expires = wrapped.expires
    if (expires and (expires <= now()))
      @remove(args)
      null
    else
      wrapped.data
